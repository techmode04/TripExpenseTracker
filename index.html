🚨 IMPORTANT RULE FOR CORRECT SETUP (Pull/Push failures stop karne ke liye):

Jab aap Apps Script deploy karein, toh yeh do options bilkul sahi select hone chahiye:

  • Execute as: Me (your-email@gmail.com)
  • Who has access: Anyone (Agur aapne "Anyone with Google Account" select kiya, toh browser request fail ho jayegi).

Code for your Apps Script:

function doGet(e) {
  var scriptProp = PropertiesService.getScriptProperties();
  var data = scriptProp.getProperty("wanderbudget_db");
  return ContentService.createTextOutput(JSON.stringify({
    status: "success",
    data: data ? JSON.parse(data) : null
  })).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var requestData = JSON.parse(e.postData.contents);
    if (requestData.action === "sync") {
      var scriptProp = PropertiesService.getScriptProperties();
      scriptProp.setProperty("wanderbudget_db", JSON.stringify(requestData.data));
      return ContentService.createTextOutput(JSON.stringify({ status: "success" }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
